12 /**
13 * The Boolean class wraps a value of the primitive type
14 * @c gbool in an object. An object of type @c Boolean contains
15 * a single field whose type is @c gbool.
16 */
17 class Boolean final : public virtual Comparable<Boolean>
18 {
19 public:
20 /**
21 * The @c Boolean object corresponding to the primitive
22 * value @c true.
23 */
24 static const Boolean TRUE;
25
26 /**
27 * The @c Boolean object corresponding to the primitive
28 * value @c false.
29 */
30 static const Boolean FALSE;
31
32 private:
33 /**
34 * The value of the Boolean.

38 public:
39 /**
40 * Allocates a @c Boolean object representing the

45 CORE_IMPLICIT Boolean(gbool value);
46
47 /**
48 * Parses the string argument as a boolean. The @c boolean
49 * returned represents the value @c true if the string argument
50 * is equal, ignoring case, to the string @c "true". Otherwise,
51 * a false value is returned.
52 *
53 * Example:
54 * @code{.cpp}
55 * Boolean::parseBoolean("True") // returns true.
56 * Boolean::parseBoolean("yes") // returns false.
57 * @endcode
58 *
59 * @param s the @c String containing the boolean representation
60 * to be parsed
61 * @return the boolean represented by the string argument.
62 */
63 static gbool parseBoolean(String const& s);
64
65 /**
66 * Returns the value of this @c Boolean object as a boolean
67 * primitive.
68 *
69 * @return the primitive @c boolean value of this object.
70 */
71 gbool booleanValue() const;
72
73 /**
74 * Returns a @c Boolean instance representing the specified
75 * @c boolean value. If the specified @c boolean value
76 * is @c true, this method returns @c Boolean::TRUE;
77 * if it is @c false, this method returns @c Boolean::FALSE.
78 *
79 * @param b a boolean value.
80 * @return a @c Boolean instance representing @c b.
81 *
82 */
83 static Boolean valueOf(gbool b);
84
85 /**
86 * Returns a @c Boolean with a value represented by the
87 * specified string. The @c Boolean returned represents a
88 * true value if the string argument is equal, ignoring case,
89 * to the string @c "true". Otherwise, a false value is returned.
90 *
91 * @param s a string.
92 * @return the @c Boolean value represented by the string.
93 */
94 static Boolean valueOf(String const& s);
95
96 /**
97 * Returns a @c String object representing the specified

99 * the string @c "true" will be returned, otherwise the
100 * string @c "false" will be returned.
101 *
102 * @param b the boolean to be converted
103 * @return the string representation of the specified @c boolean
104 *
105 */
106 static String toString(gbool b);
107
108 /**
109 * Returns a @c String object representing this Boolean's
110 * value. If this object represents the value @c true,
111 * a string equal to @c "true" is returned. Otherwise, a
112 * string equal to @c "false" is returned.
113 *
114 * @return a string representation of this object.
115 */
116 String toString() const override;
117
118 /**
119 * Returns a hash code for this @c Boolean object.
120 *
121 * @return the integer @c 1231 if this object represents
122 * @c true; returns the integer @c 1237 if this
123 * object represents @c false.
124 */
125 gint hash() const override;
126
127 /**
128 * Returns a hash code for a @c boolean value; compatible with

133 *
134 */
135 static gint hash(gbool value);
136
137 /**
138 * Returns @c true if and only if the argument is not
139 * @c null and is a @c Boolean object that
140 * represents the same @c boolean value as this object.
141 *
142 * @param obj the object to compare with.
143 * @return @c true if the Boolean objects represent the
144 * same value; @c false otherwise.
145 */
146 gbool equals(Object const& obj) const override;
147
148 /**
149 * Compares this @c Boolean instance with another.
150 *
151 * @param other the @c Boolean instance to be compared
152 * @return zero if this object represents the same boolean value as the
153 * argument; a positive value if this object represents true
154 * and the argument represents false; and a negative value if
155 * this object represents false and the argument represents true
156 *
157 */
158 gint compareTo(Boolean const& other) const override;
159
160 /**
161 * Compares two @c boolean values.
162 * The value returned is identical to what would be returned by:
163 * @code
164 * Boolean::valueOf(x).compareTo(Boolean::valueOf(y))
165 * @endcode
166 *
167 * @param x the first @c boolean to compare
168 * @param y the second @c boolean to compare
169 * @return the value @c 0 if @c x == y;
170 * a value less than @c 0 if @c !x && y; and
171 * a value greater than @c 0 if @c x && !y
172 *
173 */
174 static gint compare(gbool x, gbool y);
175
176 /**
177 * Returns the result of applying the logical AND operator to the
178 * specified @c boolean operands.
179 *
180 * @param a the first operand
181 * @param b the second operand
182 * @return the logical AND of @c a and @c b
183 *
184 */
185 static gbool logicalAnd(gbool a, gbool b);
186
187 /**
188 * Returns the result of applying the logical OR operator to the
189 * specified @c boolean operands.
190 *
191 * @param a the first operand
192 * @param b the second operand
193 * @return the logical OR of @c a and @c b
194 *
195 */
196 static gbool logicalOr(gbool a, gbool b);
197
198 /**
199 * Returns the result of applying the logical XOR operator to the
200 * specified @c boolean operands.
201 *
202 * @param a the first operand
203 * @param b the second operand
204 * @return the logical XOR of @c a and @c b
205 *
206 */
207 static gbool logicalXor(gbool a, gbool b);
208
209 /**
210 * The number of bits used to represent a @c gbyte value in two's

214
215 /**

217 * complement binary form.

219 CORE_FAST static gint BYTES = SIZE / Byte::SIZE;
