56 /**
57 * Parses the string argument as a signed @c gbyte in the
58 * radix specified by the second argument. The characters in the
59 * string must all be digits, of the specified radix (as
60 * determined by whether @c Character::digit(gchar,gint)
61 * returns a non-negative value) except that the first
62 * character may be an ASCII minus sign @c '-'('&#92;u002D')
63 * to indicate a negative value or an ASCII plus sign @c '+'('&#92;u002B') to
64 * indicate a positive value. The resulting @c gbyte value is
65 * returned.
66 *
67 * An exception of type @c NumberFormatException is
68 * thrown if any of the following situations occurs:
69 *
70 * - The first argument is a string of length zero.
71 * - The radix is either smaller than @c Character::MIN_RADIX or
72 * larger than @c Character::MAX_RADIX.
73 *
74 * - Any character of the string is not a digit of the
75 * specified radix, except that the first character may be a minus
76 * sign @c '-'('&#92;u002D') or plus sign @c '+'('&#92;u002B') provided that the
77 * string is longer than length @c 1.
78 *
79 * - The value represented by the string is not a value of type @c gbyte.
80 *
81 *
82 * @param s the @c String containing the @c gbyte
83 * representation to be parsed
84 * @param radix the radix to be used while parsing @c s
85 * @return the @c gbyte value represented by the string
86 * argument in the specified radix
87 * @throws NumberFormatException If the string does
88 * not contain a parsable @c gbyte.
89 */
90 static gbyte parseByte(String const &s, gint radix);

92 /**
93 * Parses the string argument as a signed decimal @c
94 * byte. The characters in the string must all be decimal digits,
95 * except that the first character may be an ASCII minus sign
96 * @c '-'('&#92;u002D') to indicate a negative
97 * value or an ASCII plus sign @c '+'('&#92;u002B') to indicate a positive value. The
98 * resulting @c gbyte value is returned, exactly as if the
99 * argument and the radix 10 were given as arguments to
100 * the @c parseByte(String,gint) method.
101 *
102 * @param s a @c String containing the
103 * @c gbyte representation to be parsed
104 * @return the @c gbyte value represented by the
105 * argument in decimal
106 * @throws NumberFormatException if the string does not
107 * contain a parsable @c gbyte.
108 */
109 static gbyte parseByte(String const &s);

153 /**
154 * Decodes a @c String into a @c Byte.
155 * Accepts decimal, hexadecimal, and octal numbers given by
156 * the following grammar:
157 *
158 * <li> DecodableString:
159 * <ul>
160 * <li> @a Sign <sub>(opt)</sub> DecimalNumeral
161 * <li> @a Sign <sub>(opt)</sub> @c 0x @a HexDigits
162 * <li> @a Sign <sub>(opt)</sub> @c 0X @a HexDigits
163 * <li> @a Sign <sub>(opt)</sub> @c # @a HexDigits
164 * <li> @a Sign <sub>(opt)</sub> @c 0 @a OctalDigits
165 * <li> @a Sign <sub>(opt)</sub> @c 0b @a BinaryDigits
166 * <li> @a Sign <sub>(opt)</sub> @c 0B @a BinaryDigits
167 * </ul>
168 * <li> Sign:
169 * <ul>
170 * <li> @c -
171 * <li> @c +
172 * <ul>
173 * The sequence of characters following an optional
174 * sign and/or radix specifier ( @c "0x", @c "0X",
175 * @c "#", @c "0b", @c "0B", or @a leading-zero) is parsed as by the
176 * @c Byte::parseByte method with the indicated radix (10, 16, 8 or 2).
177 * This sequence of characters must represent a positive value or
178 * a @c NumberFormatException will be thrown. The result is
179 * negated if first character of the specified @c String is
180 * the minus sign. No whitespace characters are permitted in the
181 * @c String.
182 *
183 * @param nm the @c String to decode.
184 * @return A @c Byte object holding the @c gbyte value represented by @c nm
185 * @throws NumberFormatException if the @c String does not contain a parsable @c gbyte.
186 */
187 static Byte decode(String const &nm);

320 /**
321 * Converts the argument to an @c gint by an unsigned
