99 /**
100 * Returns the cause of this throwable or @a itself if the
101 * cause is nonexistent or unknown. (The cause is the throwable that
102 * caused this throwable to get thrown.)
103 *
104 * This implementation returns the cause that was supplied via one of
105 * the constructors requiring a @c Throwable, or that was set after
106 * creation. While it is typically unnecessary to override this method, a subclass can override

114 * @return the cause of this throwable or @a itself if the

119 /**
120 * Returns a short description of this throwable.
121 * The result is the concatenation of:
122 * @li the @a name of the class of this object
123 * @li @c ": " (a colon and a space)
124 * @li the result of invoking this object's @c localizedMessage() method

126 * If @c localizedMessage returns @c "", then just the class name

140 /**

143 * @note this method should be called after call of @c throws method, otherwise it thrown

148 CORE_NORETURN void rethrow() const;

150 protected:
151 /**

153 *
154 * Example: @code
155 * class MyClass: public virtual Exception

161 * // or throw *this; // (deprecated)

163 * };
164 * @endcode
165 *
166 * @note All subclass must be override this method to
167 * preseve perfect chaining and this method use
168 * the c++ native @c throw for compatibity with
169 * standard exception.
170 */
171 CORE_NORETURN virtual void selfThrow() const = 0;
172
173 private:
174 /**

181 */
182 ByteStream what() const CORE_NOTHROW CORE_FINAL;
