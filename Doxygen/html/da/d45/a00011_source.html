130 /**
131 * Obtain newly created @c ByteArray instance with primitive byte array,
132 * such as @c int8_t[], @c int32_t[], @c int64_t[], @c int128_t[] (signed and unsigned)
133 *
134 * @note if the memory size of element type is greater than 1 byte, the value will be truncate
135 * 1 byte, by narrowing convertion.
136 *
137 * @tparam T The array type
138 * @tparam E The byte type
139 * @param array The array to be copied
140 * @return The new ByteArray that contains all values of given array
141 */
142 template <class T,
143 Class<gbool>::OnlyIf<Class<T>::isArray()> = true,
144 class E = typename Class<T>::ArrayElement,
145 Class<gbool>::OnlyIf<Class<E>::isInteger() || Class<E>::isCharacter() && sizeof(E) == 1> = true>
146 static ByteArray copyOf(T &&array)
147 {
148 CORE_FAST glong n = Class<T>::MEMORY_SIZE / Class<E>::MEMORY_SIZE;
149
150 switch (n)
151 {
152 case 0:
153 return of();
154 case 1:
155 return of(array[0]);
156 case 2:
157 return of(array[0], array[1]);
158 case 3:
159 return of(array[0], array[1], array[2]);
160 case 4:
161 return of(array[0], array[1], array[2], array[3]);
162 case 5:
163 return of(array[0], array[1], array[2], array[3], array[4]);
164 case 6:
165 return of(array[0], array[1], array[2], array[3], array[4],
166 array[5]);
167 case 7:
168 return of(array[0], array[1], array[2], array[3], array[4],
169 array[5], array[6]);
170 case 8:
171 return of(array[0], array[1], array[2], array[3], array[4],
172 array[5], array[6], array[7]);
173 case 9:
174 return of(array[0], array[1], array[2], array[3], array[4],
175 array[5], array[6], array[7], array[8]);
176 case 10:
177 return of(array[0], array[1], array[2], array[3], array[4],
178 array[5], array[6], array[7], array[8], array[9]);
179 default:
180 {
181 CORE_FAST gint length = n > SOFT_MAX_LENGTH ? SOFT_MAX_LENGTH : (gint)n;
182
183 ByteArray bytes = ByteArray(length);
184
185 for (int i = 0; i < length; ++i)
186 {
187 bytes.set(i, array[i]);
188 }
189
190 return CORE_CAST(ByteArray &&, bytes);
191 }
192 }
193 }
