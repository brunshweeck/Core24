171 /**
172 * Returns a string representation of this stack trace element.
173 *
174 * @apiNote The format of this string depends on the implementation, but the
175 * following examples may be regarded as typical:
176 * @li @c "foo@9.0/com::foo::Main.run(Main.cpp:101)"
177 * - See the description below.
178 * @li @c "com.foo.loader/foo@9.0/com::foo::Main.run(Main.cpp)"
179 * - The line number is unavailable.
180 * @li @c "foo@9.0/com.foo.Main.run(?)"
181 * - Neither the file name nor the line number is available.
182 * @li @c "foo@9.0/com::foo::Main.run(#)"
183 * - The method containing the execution point is a native method.
184 * @li @c "com::foo::bar::App.run(App.cpp:12)"
185 * - The class of the execution point is defined in the unnamed module.
186 * @li @c "acme@2.1/org::acme.Lib.test(Lib.cpp:80)"
187 * - The class of the execution point is defined in @c acme module.
188 * @li @c "MyClass.mash(MyClass.cpp:9)"
189 * - @c MyClass class is on the application class path.
190 *
191 * The first example shows a stack trace element consisting of
192 * three elements, each separated by @c "/", followed by
193 * the source file name and the line number of the source line
194 * containing the execution point.
195 *
196 * The first element @c "foo@9.0" is the module name and version.
197 * The seconds element is the method containing the execution point;
198 * @c "com.foo.Main"" is the fully-qualified class name and "@c run"
199 * is the name of the method. @c "Main.cpp" is the source file name
200 * and @c "101" is the line number.
201 *
202 * If a class is defined in an <em>unnamed module</em>
203 * then the second element is omitted as shown in
204 * @c "com::foo::bar::App.run(App.cpp:12)".
205 *
206 * If the module is an unnamed module, the second element
207 * and its following @c "/" are also omitted as shown in
208 * @c "MyClass.mash(MyClass.cpp:9)".
209 *
210 * The @c toString method may return two different values on two
211 * @c Trace instances that are @a equal, for example one created via the
212 * constructor, and one obtained from @c Throwable, where an implementation may
213 * choose to omit some element in the returned string.
214 */
215 String toString() const;

217 /**
218 * Returns true if the specified object is another @c Trace instance
219 * representing the same execution point as this instance. Two stack
220 * trace @c a and @c b are equal if and only if:
221 * @code
222 * a.moduleName().equals(b.moduleName()) &&
223 * a.moduleVersion().equals(b.moduleVersion()) &&
224 * a.className().equals(b.className()) &&
225 * a.methodName().equals(b.methodName()) &&
226 * a.fileName().equals(b.fileName()) &&
227 * a.lineNumber() == b.lineNumber()
228 * @endcode
229 *
230 * @param obj the object to be compared with this stack trace element.
231 * @return @c true if the specified object is another @c Trace instance
232 * representing the same execution point as this instance.
233 */
234 gbool equals(Object const& obj) const override;
